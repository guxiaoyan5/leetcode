package main

func singleNumber(nums []int) []int {
	m := map[int]int{}
	var result []int
	for _, num := range nums {
		if _, ok := m[num]; ok {
			m[num] += 1
		} else {
			m[num] = 1
		}
	}
	for key, _ := range m {
		if m[key] == 1 {
			result = append(result, key)
		}
	}
	return result
}
